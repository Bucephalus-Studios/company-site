124 inline WrapCut findWrapCut(const std::string_view & line, size_t maxWidth)
125 {
126 if(line.empty())
127 {
128 return { 0, 0 };
129 }
130
131 size_t width = 0;
132 size_t lastSpaceStart = std::string_view::npos; // byte offset where the last-seen space starts
133 size_t lastSpaceEnd = std::string_view::npos; // byte offset right after that space
134
135 auto it = line.begin();
136 const auto end = line.end();
137
138 while(it != end)
139 {
140 auto codepointStart = it;
141 size_t startOffset = static_cast<size_t>(codepointStart - line.begin());
142 utf8::utfchar32_t codepoint = utf8::next(it, end); // advances it past this codepoint
143 size_t afterOffset = static_cast<size_t>(it - line.begin());
144
145 size_t codepointWidth = codepointDisplayWidth(codepoint);
146 if(width + codepointWidth > maxWidth)
147 {
148 // If the codepoint that just overflowed is itself a space (and isn't the
149 // very first codepoint - see the forward-progress guarantee below), everything
150 // before it fit exactly, and the space itself is dropped from the output when
151 // used as a break point (see nextStart above) - so it doesn't need to "fit" to
152 // be a valid, in fact ideal, break point right at the boundary.
153 if(codepoint == U' ' && startOffset > 0)
154 {
155 return { startOffset, afterOffset };
156 }
157 if(lastSpaceStart != std::string_view::npos)
158 {
159 return { lastSpaceStart, lastSpaceEnd };
160 }
161 // No usable space - force-break, but guarantee at least one codepoint of
162 // forward progress (see doc comment above).
163 size_t cut = (startOffset == 0) ? afterOffset : startOffset;
164 return { cut, cut };
165 }
166
167 width += codepointWidth;
168 if(codepoint == U' ')
169 {
170 lastSpaceStart = startOffset;
171 lastSpaceEnd = afterOffset;
172 }
173 }
174
175 return { line.length(), line.length() };
176 }

462 inline std::vector<std::string> separate( const std::string_view & str,
463 const char separator = ',',
464 const bool omitEmptyStrings = true )
465 {
466 //If we are given an empty string, just return an empty vector
467 if(str.empty())
468 {
469 return {};
470 }
471
472 //Initialize all the variables we'll be using in our loop
473 std::vector<std::string> separatedStringsVec;
474 std::string separatedString;
475 size_t currSeparatorIndex;
476 size_t prevSeparatorIndex = 0;
477
478 //Loop until we've found all of the separator chars in the string_view
479 while(true)
480 {
481 //Look for a separator char in the string_view
482 currSeparatorIndex = str.find(separator, prevSeparatorIndex);
483 //If the separator is not found, just get everything up until the end
484 if(currSeparatorIndex == std::string_view::npos)
485 {
486 separatedString = str.substr(prevSeparatorIndex, str.length());
487 separatedStringsVec.push_back(separatedString);
488 break;
489 }
490 //Once we've found the index, of the separator char, get all of the std::string that we jumped over since the last separator character
491 separatedString = str.substr(prevSeparatorIndex, currSeparatorIndex-prevSeparatorIndex);
492 //Push it bank into our vector
493 separatedStringsVec.push_back(separatedString);
494 //Set the prevSeparatorIndex to currSeparatorIndex now
495 prevSeparatorIndex = currSeparatorIndex + 1;
496 }
497
498 //If we are omitting empty strings in our result, erase all of that we found
499 if(omitEmptyStrings)
500 {
501 separatedStringsVec.erase(std::remove(separatedStringsVec.begin(), separatedStringsVec.end(), ""), separatedStringsVec.end());
502 }
503
504 return separatedStringsVec;
505 }

528 inline std::vector<std::string> separate( const std::string_view & str,
529 const std::string_view & separator,
530 const bool omitEmptyStrings = true )
531 {
532 //If we have a single character separator, we can use a more efficient method
533 if(separator.length() == 1)
534 {
535 return stevensStringLib::separate( str, separator[0], omitEmptyStrings );
536 }
537
538 //If separator is empty, split into individual characters (codepoints, not bytes - a raw
539 //byte-by-byte split would shred a multi-byte character, e.g. Cyrillic/CJK, into invalid
540 //fragments). Walks the original UTF-8 bytes once via utf8::next() and slices each
541 //codepoint's byte range directly out of str - no need to decode into a std::u32string
542 //and re-encode each codepoint back to UTF-8, since we never need the decoded codepoint
543 //value itself, only where each character starts and ends.
544 if(separator.empty())
545 {
546 std::vector<std::string> separatedStrings;
547 separatedStrings.reserve(str.length()); // upper bound - codepoints <= bytes
548
549 auto it = str.begin();
550 const auto end = str.end();
551 while(it != end)
552 {
553 auto charStart = it;
554 utf8::next(it, end); // advances it past one codepoint
555 separatedStrings.emplace_back(charStart, it);
556 }
557 return separatedStrings;
558 }
559
560 std::vector<std::string> separatedStrings;
561 separatedStrings.reserve(8); // Reasonable default to avoid vector reallocations
562
563 size_t start = 0;
564 size_t pos = 0;
565
566 // Direct search without building intermediate strings
567 while((pos = str.find(separator, start)) != std::string_view::npos)
568 {
569 if(pos > start || !omitEmptyStrings)
570 {
571 // Construct std::string directly in vector memory - no temporary objects
572 separatedStrings.emplace_back(str.data() + start, pos - start);
573 }
574 start = pos + separator.length();
575 }
576
577 // Handle final segment
578 if(start < str.length() || (!omitEmptyStrings && start == str.length()))
579 {
580 separatedStrings.emplace_back(str.data() + start, str.length() - start);
581 }
582
583 return separatedStrings;
584 }

598 inline std::string join( const std::vector<std::string> & vec,
599 const std::string & separator,
600 const bool omitEmptyStrings = true )
601 {
602 //If an empty vector is passed in for vec, return the empty string.
603 if(vec.size() == 0)
604 {
605 return "";
606 }
607
608 //The std::string we will return
609 std::string str = "";
610 bool first = true;
611
612 //Concatenate the vector together element-by-element
613 for(int i = 0; i < vec.size(); i++)
614 {
615 //Check to see if the element at the current index is empty and if we're omitting empty strings. If so, we'll skip to the next index.
616 if(vec[i].empty() && omitEmptyStrings)
617 {
618 continue;
619 }
620 //Add separator before non-first elements
621 if(!first)
622 {
623 str += separator;
624 }
625 //For each element in the vector, concatenate it onto the end of the std::string
626 str += vec[i];
627 first = false;
628 }
629
630 //Return the joined vector of strings
631 return str;
632 }

790 inline bool isStandardNumber( const std::string_view & str )
791 {
792 //If our std::string is empty, return false
793 if(str.empty())
794 {
795 return false;
796 }
797 //If we have a std::string of length 1, just check to see if the single character is a digit
798 else if(str.length() == 1)
799 {
800 return std::isdigit(str[0]);
801 }
802 //Get the decimal point for our locale
803 char point = std::use_facet< std::numpunct<char> >(std::cout.getloc()).decimal_point();
804 bool seenDecimalPoint = false;
805 bool seenNonZeroDigit = false;
806 //Otherwise, iterate through all of the characters in the std::string and examine each one
807 for(int i = 0; i < str.length(); i++)
808 {
809 switch(i)
810 {
811 //Check to see if the first character in the std::string is a sign. We allow this!
812 case 0:
813 if(str[i] == '-' || str[i] == '+')
814 {
815 continue;
816 }
817 default:
818 //Check to see if the character is a decimal point
819 if(str[i] == point)
820 {
821 if(!seenDecimalPoint)
822 {
823 seenDecimalPoint = true;
824 }
825 //Numbers in standard notation only contain a single decimal point, so we return false.
826 else
827 {
828 return false;
829 }
830 }
831 //Check to see if the character is a numerical digit
832 else if(!std::isdigit(str[i]))
833 {
834 //If we have a character that's not a numerical digit which is not a decimal point or negative sign, it's not in standard notation. Return false.
835 return false;
836 }
837 //Check to see if the digit is nonzero
838 else if(str[i] != '0')
839 {
840 seenNonZeroDigit = true;
841 }
842 break;
843 }
844 }
845
846 if((!seenNonZeroDigit) && (stevensStringLib::startsWith(str, "+") || stevensStringLib::startsWith(str, "-")))
847 {
848 //000.000 should not be written with a + or - sign out front in standard notation.
849 return false;
850 }
851
852 //We've made it past all of the checks by this point, so we must have a std::string representing a number in standard notation.
853 return true;
854 }

911 //Make sure that we haven't reached the end of the std::string already

914 //Looks like we only have a decimal part and an 'x','X','*','e', or an 'E'.
915 return false;
916 }
917 //Get the other part of the potential scientific number
918 std::string tenToThePowerPart;
919 tenToThePowerPart = str.substr(endOfDecimalPartIndex + 1);
920 //Based on the endOfDecimalPartChar, decide how to parse the tenToThePowerPart should be parsed

922 str[endOfDecimalPartIndex] == 'X' ||

933 //We don't have a "10^" included in our tenToThePowerPart, so it can't be scientific notation.
934 return false;
935 }
936 //Now the rest of the tenToThePower part after 10^ must be an integer
937 if(!stevensStringLib::isInteger(tenToThePowerPart.substr(3)))
938 {
939 //The rest of the tenToThePower part is not a valid integer
940 return false;
941 }
942 }
943 else if( str[endOfDecimalPartIndex] == 'e' ||
944 str[endOfDecimalPartIndex] == 'E' )
945 {
946 //We are expecting to read an integer
947 if(!stevensStringLib::isInteger(tenToThePowerPart))
948 {
949 //The rest of the tenToThePower part is not a valid integer
950 return false;
951 }
952 }
953
954 //All checks have been passed. The number is in scientific notation!
955 return true;
956 }

1233 size_t firstLineWidth = std::string::npos )
1234 {

1238 return "";
1239 }
1240 if(firstLineWidth == std::string::npos)

1661 inline std::string format( std::string str,
1662 const std::vector<std::string> & replaceStrs,
1663 const char & openingFormatBrace = '{',
1664 const char & closingFormatBrace = '}' )
1665 {
1666 size_t startPos = 0; //The position in the std::string of the opening brace we find
1667 size_t endPos = 0; //The position in the std::string of the closing brace we find
1668 std::string braceContents; //Contents of the formatting braces that we find
1669 size_t replaceIndex = 0; //The index of the replaceStrs vector that we'll use to replace a format-braced substring
1670
1671 //Look for an opening format brace
1672 while ((startPos = str.find(openingFormatBrace, startPos)) != std::string::npos)
1673 {
1674 //When we find an opening format brace, look for the closing format brace
1675 endPos = str.find(closingFormatBrace, startPos);
1676 if (endPos != std::string::npos)
1677 {
1678 //Check to see what's inside the formatting braces
1679 braceContents = str.substr(startPos + 1, endPos - startPos - 1);
1680 if(!isNumber(braceContents))
1681 {
1682 //If it's not a number, move on
1683 startPos++;
1684 continue;
1685 }
1686 //Find the correct replaceStr from our vector to replace the format-braced substring
1687 replaceIndex = std::stoi(braceContents);
1688 if ( (replaceIndex < replaceStrs.size()) && (replaceIndex >= 0) )
1689 {
1690 //Make the replacement
1691 str.replace(startPos, endPos - startPos + 1, replaceStrs[replaceIndex]);
1692 startPos += replaceStrs[replaceIndex].size();
1693 }
1694 else
1695 {
1696 //If the index of replaceStrs does not exist to replace the current format-braced substring, skip it.
1697 startPos++;
1698 }
1699 }
1700 //If we can't find a closing format brace, stop executing the function, as we know there won't be any more replacements we can make
1701 else
1702 {
1703 break;
1704 }
1705 }
1706
1707 return str;
1708 }
